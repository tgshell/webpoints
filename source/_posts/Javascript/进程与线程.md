### 执行顺序

```
const startTime = new Date().getTime();
setTimeout(() => {
	console.log('settimeout end', new Date().getTime() - startTime);
}, 100);
for (let i = 0; i < 50000; i ++) {
	for (let j = 0; j < 50000; j ++) {
		i + j;
    }
}
console.log('script end', new Date().getTime() - startTime);
```
